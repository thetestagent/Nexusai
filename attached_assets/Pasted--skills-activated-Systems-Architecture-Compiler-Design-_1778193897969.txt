```rust
use miette::{Diagnostic, NamedSource, SourceSpan};
use thiserror::Error;

/// Represents a location in source code.
#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
pub struct Span {
    pub start: usize,
    pub end: usize,
}

impl Span {
    pub fn new(start: usize, end: usize) -> Self {
        Self { start, end }
    }

    pub fn merge(self, other: Span) -> Span {
        Span {
            start: self.start.min(other.start),
            end: self.end.max(other.end),
        }
    }

    pub fn len(&self) -> usize {
        self.end - self.start
    }

    pub fn is_empty(&self) -> bool {
        self.start == self.end
    }
}

impl From<Span> for SourceSpan {
    fn from(span: Span) -> Self {
        SourceSpan::new(span.start.into(), span.len().into())
    }
}

/// Core error types for the NEXUS compiler pipeline.
#[derive(Debug, Error, Diagnostic)]
pub enum NexusError {
    // ── Lexer Errors ──
    #[error("Unexpected character '{}'", .found)]
    #[diagnostic(code(nexus::lexer::unexpected_char))]
    UnexpectedCharacter {
        found: char,
        #[label("here")]
        span: SourceSpan,
    },

    #[error("Unterminated string literal")]
    #[diagnostic(code(nexus::lexer::unterminated_string))]
    UnterminatedString {
        #[label("starts here")]
        span: SourceSpan,
    },

    // ── Parser Errors ──
    #[error("Unexpected token: expected {expected}, found {found}")]
    #[diagnostic(code(nexus::parser::unexpected_token))]
    UnexpectedToken {
        expected: String,
        found: String,
        #[label("unexpected {found}")]
        span: SourceSpan,
    },

    #[error("Missing semicolon after statement")]
    #[diagnostic(code(nexus::parser::missing_semicolon))]
    MissingSemicolon {
        #[label("expected ';' here")]
        span: SourceSpan,
    },

    // ── Semantic / Type Errors ──
    #[error("Type mismatch: cannot assign {found} to {expected}")]
    #[diagnostic(code(nexus::semantic::type_mismatch))]
    TypeMismatch {
        expected: String,
        found: String,
        #[label("{found} is not assignable to {expected}")]
        span: SourceSpan,
    },

    // ── Engineering Unit Errors (spec §19) ──
    #[error("Unit mismatch: cannot assign {found} to {expected}")]
    #[diagnostic(code(nexus::engineering::unit_mismatch))]
    UnitMismatch {
        expected: String,
        found: String,
        #[label("incompatible units: {found} vs {expected}")]
        span: SourceSpan,
    },

    #[error("Dimensional error: {msg}")]
    #[diagnostic(code(nexus::engineering::dimensional))]
    DimensionalError {
        msg: String,
        #[label("dimensional inconsistency")]
        span: SourceSpan,
    },

    // ── Undefined Reference ──
    #[error("Undefined identifier: '{name}'")]
    #[diagnostic(code(nexus::semantic::undefined))]
    UndefinedIdentifier {
        name: String,
        #[label("'{name}' is not defined in this scope")]
        span: SourceSpan,
    },

    // ── AI Entity Errors ──
    #[error("Invalid entity definition: {msg}")]
    #[diagnostic(code(nexus::ai::invalid_entity))]
    InvalidEntityDefinition {
        msg: String,
        #[label("entity definition error")]
        span: SourceSpan,
    },

    // ── Internal / ICE ──
    #[error("Internal compiler error: {msg}")]
    #[diagnostic(code(nexus::internal))]
    InternalError {
        msg: String,
        #[label("this is a bug in the NEXUS compiler")]
        span: Option<SourceSpan>,
    },
}

/// Result type used throughout the NEXUS compiler.
```

```rust
/// All token types in the NEXUS language (§13).
#[derive(Debug, Clone, PartialEq, Logos)]
#[logos(skip r"[ \t\n\r\f]+")]  // skip whitespace
#[logos(skip r"//[^\n]*")]      // skip line comments
pub enum TokenKind {
    // ── Keywords ──
    #[token("let")]       Let,
    #[token("entity")]    Entity,
    #[token("equation")]  Equation,

    #[token("fn")]        Fn,
    #[token("return")]    Return,

    #[token("impl")]      Impl,
    #[token("match")]     Match,
    #[token("enum")]      Enum,
    #[token("true")]      True,
    #[token("false")]     False,

    #[token("+")]  Plus,
    #[token("-")]  Minus,
    #[token("*")]  Star,
    #[token("/")]  Slash,

    #[token("}")]  RBrace,

    pub span: Span,
    pub lexeme: String,

/// The primary lexer function: source text → token stream.
pub struct Lexer<'source> {
    source: &'source str,
    inner: LogosLexer<'source, TokenKind>,
    string_buf: String,
    in_string: bool,
    string_start: usize,
}

impl<'source> Lexer<'source> {
    pub fn new(source: &'source str) -> Self {
        Self {
            source,
            inner: TokenKind::lexer(source),
            string_buf: String::new(),
            in_string: false,
            string_start: 0,
        }
    }

    /// Lex the entire source and return all tokens.
    pub fn tokenize(&mut self) -> NexusResult<Vec<Token>> {
        let mut tokens = Vec::new();

        while let Some(kind) = self.inner.next() {
            let span_range = self.inner.span();
            let span = Span::new(span_range.start, span_range.end);

            match kind {
                TokenKind::Error => {
                    let bad_char = self.source[span_range.start..]
                        .chars()
                        .next()
                        .unwrap_or('?');
                    return Err(NexusError::UnexpectedCharacter {
                        found: bad_char,
                        span: span.into(),
                    });
                }

                TokenKind::StringStart => {
                    self.in_string = true;
                    self.string_start = span_range.start;
                    self.string_buf.clear();
                    // Continue consuming characters until closing quote
                    let remaining = &self.source[span_range.end..];
                    if let Some(end_pos) = remaining.find('"') {
                        let content = &remaining[..end_pos];
                        self.string_buf.push_str(content);
                        // Advance the lexer past the string content and closing quote
                        let skip_len = end_pos + 1; // +1 for closing quote
                        // We need to manually bump the lexer's position
                        // Logos doesn't directly support this, so we'll handle it differently
                        // For now, we'll store the content
                        let span = Span::new(self.string_start, span_range.end + skip_len);
                        tokens.push(Token::new(
                            TokenKind::Identifier, // placeholder — we'll refine string token
                            span,
                            format!("\"{}\"", self.string_buf),
                        ));
                    } else {
                        return Err(NexusError::UnterminatedString {
                            span: Span::new(self.string_start, self.source.len()).into(),
                        });
                    }
                    self.in_string = false;
                }

                _ => {
                    let lexeme = &self.source[span_range.start..span_range.end];
                    tokens.push(Token::new(kind, span, lexeme));
                }
            }
        }

        Ok(tokens)
    }
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn test_let_binding() {
        let source = "let voltage = 12V";
        let mut lexer = Lexer::new(source);
        let tokens = lexer.tokenize().unwrap();
        let kinds: Vec<TokenKind> = tokens.iter().map(|t| t.kind.clone()).collect();
        assert_eq!(kinds, vec![
            TokenKind::Let,
            TokenKind::Identifier,
            TokenKind::Equals,
            TokenKind::UnitLiteral,

        let kinds: Vec<TokenKind> = tokens.iter().map(|t| t.kind.clone()).collect();

    }
```

```rust
    bump: &'arena Bump,
}

impl<'arena> AstArena<'arena> {
    pub fn new(bump: &'arena Bump) -> Self {
        Self { bump }
    }

    /// Allocate any AST node on the arena and return a reference.
    pub fn alloc<T>(&self, value: T) -> &'arena mut T {
        self.bump.alloc(value)
    }
}
```

The parser converts the token stream into AST nodes. I'm using a recursive descent parser for clarity and precise error recovery.

```toml
[dependencies]
```

crates/nexus-parser/src/lib.rs

```rust
use bumpalo::Bump;
use nexus_ast::*;
use nexus_errors::{NexusError, NexusResult, Span};
use nexus_lexer::{Lexer, Token, TokenKind};
use std::iter::Peekable;
use std::vec::IntoIter;

pub struct Parser<'source> {
    tokens: Peekable<IntoIter<Token>>,
    source: &'source str,
    arena: &'source Bump,
}

impl<'source> Parser<'source> {
    pub fn new(source: &'source str, arena: &'source Bump) -> NexusResult<Self> {
        let mut lexer = Lexer::new(source);
        let tokens = lexer.tokenize()?;
        Ok(Self {
            tokens: tokens.into_iter().peekable(),
            source,
            arena,
        })
    }

    /// Parse an entire NEXUS program.
    pub fn parse_program(&mut self) -> NexusResult<Program> {
        let mut declarations = Vec::new();

        while self.peek().is_some() {
            let decl = self.parse_declaration()?;
            declarations.push(decl);
        }

        Ok(Program::new(declarations))
    }

    // ── Declaration Parsing ──

    fn parse_declaration(&mut self) -> NexusResult<AstNode> {
        match self.peek().map(|t| t.kind.clone()) {
            Some(TokenKind::Let) => self.parse_let_binding(),
            Some(TokenKind::Equation) => self.parse_equation_def(),
            Some(TokenKind::Entity) => self.parse_entity_def(),
            Some(TokenKind::Sensor) => self.parse_sensor_def(),
            Some(_) => self.parse_expression(),
            None => Err(self.unexpected_eof()),
        }
    }

    fn parse_let_binding(&mut self) -> NexusResult<AstNode> {
        let let_token = self.expect(TokenKind::Let)?;
        let ident = self.expect_identifier()?;
        let type_annotation = if self.match_token(TokenKind::Colon) {
            Some(Box::new(self.parse_type_annotation()?))
        } else {
            None
        };
        self.expect(TokenKind::Equals)?;
        let value = self.parse_expression()?;
        self.expect_semicolon()?;

        let span = Span::merge(let_token.span, self.last_span());

        Ok(AstNode::LetBinding(LetBinding {
            id: NodeId::new(),
            span,
            name: ident.lexeme.clone(),
            type_annotation,
            value: Box::new(value),
        }))
    }

    fn parse_equation_def(&mut self) -> NexusResult<AstNode> {
        let eq_token = self.expect(TokenKind::Equation)?;
        let name_token = self.expect_identifier()?;
        self.expect(TokenKind::LBrace)?;
        let body = self.parse_expression()?;
        self.expect(TokenKind::RBrace)?;

        let span = Span::merge(eq_token.span, self.last_span());

        Ok(AstNode::EquationDef(EquationDef {
            id: NodeId::new(),
            span,
            name: name_token.lexeme.clone(),
            body: Box::new(body),
        }))
    }

    fn parse_entity_def(&mut self) -> NexusResult<AstNode> {
        let entity_token = self.expect(TokenKind::Entity)?;
        let name_token = self.expect_identifier()?;
        self.expect(TokenKind::LBrace)?;

        let mut memory = None;
        let mut skills = Vec::new();
        let mut reactions = Vec::new();

        while !self.check(TokenKind::RBrace) {
            match self.peek().map(|t| t.kind.clone()) {
                Some(TokenKind::Memory) => {
                    self.advance();
                    memory = Some(self.parse_memory_kind()?);
                }
                Some(TokenKind::Skill) => {
                    skills.push(self.parse_skill_def()?);
                }
                Some(TokenKind::React) => {
                    reactions.push(self.parse_react_block()?);
                }
                _ => {
                    return Err(self.unexpected_token("memory, skill, or react"));
                }
            }
        }

        self.expect(TokenKind::RBrace)?;

        let span = Span::merge(entity_token.span, self.last_span());

        Ok(AstNode::EntityDef(EntityDef {
            id: NodeId::new(),
            span,
            name: name_token.lexeme.clone(),
            memory,
            skills,
            reactions,
        }))
    }

    fn parse_sensor_def(&mut self) -> NexusResult<AstNode> {
        let sensor_token = self.expect(TokenKind::Sensor)?;
        let name_token = self.expect_identifier()?;
        self.expect(TokenKind::LBrace)?;

        // Parse pin
        // Simplified: expect identifier for pin name, then optional frequency
        let _pin = self.expect_identifier()?;
        let mut frequency = None;

        if self.match_token(TokenKind::Identifier) {
            // Expect frequency keyword and value
            if self.peek().map(|t| t.lexeme.as_str()) == Some("frequency") {
                self.advance();
                let freq_literal = self.parse_unit_literal()?;
                match freq_literal {
                    AstNode::UnitLiteral(ul) => frequency = Some(ul),
                    _ => {}
                }
            }
        }

        self.expect(TokenKind::RBrace)?;

        let span = Span::merge(sensor_token.span, self.last_span());

        Ok(AstNode::SensorDef(SensorDef {
            id: NodeId::new(),
            span,
            name: name_token.lexeme.clone(),
            pin: String::new(), // placeholder
            frequency,
        }))
    }

    // ── Expression Parsing (Pratt parser for operator precedence) ──

    fn parse_expression(&mut self) -> NexusResult<AstNode> {
        self.parse_binary_expr(0)
    }

    fn parse_binary_expr(&mut self, min_prec: u8) -> NexusResult<AstNode> {
        let mut left = self.parse_primary()?;

        while let Some(op) = self.peek_operator() {
            let prec = op_precedence(&op);
            if prec < min_prec {
                break;
            }
            let op_token = self.advance().unwrap();
            let op = token_to_binary_op(&op_token.kind).unwrap();
            let right = self.parse_binary_expr(prec + 1)?;

            let span = Span::merge(left.span(), right.span());
            left = AstNode::BinaryOp(BinaryOp {
                id: NodeId::new(),
                span,
                op,
                left: Box::new(left),
                right: Box::new(right),
            });
        }

        Ok(left)
    }

    fn parse_primary(&mut self) -> NexusResult<AstNode> {
        match self.peek().map(|t| t.kind.clone()) {
            Some(TokenKind::UnitLiteral) => self.parse_unit_literal(),
            Some(TokenKind::IntLiteral) => self.parse_int_literal(),
            Some(TokenKind::FloatLiteral) => self.parse_float_literal(),
            Some(TokenKind::Identifier) => self.parse_identifier_or_call(),
            Some(TokenKind::LParen) => {
                self.advance();
                let expr = self.parse_expression()?;
                self.expect(TokenKind::RParen)?;
                Ok(expr)
            }
            Some(TokenKind::LBrace) => self.parse_block(),
            _ => Err(self.unexpected_token("expression")),
        }
    }

    fn parse_unit_literal(&mut self) -> NexusResult<AstNode> {
        let token = self.expect(TokenKind::UnitLiteral)?;
        let lexeme = &token.lexeme;

        // Parse "12V" into value=12, unit=Volt
        let (value_str, unit_str): (&str, &str) = lexeme
            .split_at(lexeme.find(|c: char| !c.is_digit(10) && c != '.').unwrap_or(lexeme.len()));

        let value: f64 = value_str.parse().map_err(|_| {
            NexusError::InternalError {
                msg: format!("Failed to parse unit value: {}", value_str),
                span: Some(token.span.into()),
            }
        })?;

        let unit = EngineeringUnit::from_str(unit_str).ok_or_else(|| {
            NexusError::InternalError {
                msg: format!("Unknown unit: {}", unit_str),
                span: Some(token.span.into()),
            }
        })?;

        Ok(AstNode::UnitLiteral(UnitLiteral {
            id: NodeId::new(),
            span: token.span,
            value,
            unit,
        }))
    }

    fn parse_int_literal(&mut self) -> NexusResult<AstNode> {
        let token = self.expect(TokenKind::IntLiteral)?;
        let value: i64 = token.lexeme.parse().map_err(|_| {
            NexusError::InternalError {
                msg: format!("Invalid integer: {}", token.lexeme),
                span: Some(token.span.into()),
            }
        })?;

        Ok(AstNode::IntLiteral(IntLiteral {
            id: NodeId::new(),
            span: token.span,
            value,
        }))
    }

    fn parse_float_literal(&mut self) -> NexusResult<AstNode> {
        let token = self.expect(TokenKind::FloatLiteral)?;
        let value: f64 = token.lexeme.parse().map_err(|_| {
            NexusError::InternalError {
                msg: format!("Invalid float: {}", token.lexeme),
                span: Some(token.span.into()),
            }
        })?;

        Ok(AstNode::FloatLiteral(FloatLiteral {
            id: NodeId::new(),
            span: token.span,
            value,
        }))
    }

    fn parse_identifier_or_call(&mut self) -> NexusResult<AstNode> {
        let token = self.expect_identifier()?;
        Ok(AstNode::Identifier(Identifier {
            id: NodeId::new(),
            span: token.span,
            name: token.lexeme.clone(),
        }))
    }

    fn parse_block(&mut self) -> NexusResult<AstNode> {
        let start = self.expect(TokenKind::LBrace)?;
        let mut statements = Vec::new();

        while !self.check(TokenKind::RBrace) {
            statements.push(self.parse_declaration()?);
        }

        let end = self.expect(TokenKind::RBrace)?;
        let span = Span::merge(start.span, end.span);

        Ok(AstNode::Block(Block {
            id: NodeId::new(),
            span,
            statements,
        }))
    }

    // ── AI Parsing ──

    fn parse_memory_kind(&mut self) -> NexusResult<MemoryKind> {
        match self.peek().map(|t| t.kind.clone()) {
            Some(TokenKind::Persistent) => {
                self.advance();
                Ok(MemoryKind::Persistent)
            }
            _ => Ok(MemoryKind::Transient),
        }
    }

    fn parse_skill_def(&mut self) -> NexusResult<SkillDef> {
        let skill_token = self.expect(TokenKind::Skill)?;
        let name_token = self.expect_identifier()?;
        let body = match self.parse_block()? {
            AstNode::Block(b) => b,
            other => {
                return Err(NexusError::InternalError {
                    msg: "Expected block for skill body".into(),
                    span: Some(other.span().into()),
                })
            }
        };

        Ok(SkillDef {
            id: NodeId::new(),
            span: Span::merge(skill_token.span, body.span),
            name: name_token.lexeme.clone(),
            body,
        })
    }

    fn parse_react_block(&mut self) -> NexusResult<ReactBlock> {
        let react_token = self.expect(TokenKind::React)?;
        let trigger = self.parse_expression()?;
        let body = Box::new(self.parse_block()?);
        let span = Span::merge(react_token.span, body.span());

        Ok(ReactBlock {
            id: NodeId::new(),
            span,
            trigger,
            body,
        })
    }

    fn parse_type_annotation(&mut self) -> NexusResult<AstNode> {
        self.parse_primary()
    }

    // ── Helpers ──

    fn peek(&mut self) -> Option<&Token> {
        self.tokens.peek()
    }

    fn advance(&mut self) -> Option<Token> {
        self.tokens.next()
    }

    fn check(&mut self, kind: TokenKind) -> bool {
        self.peek().map(|t| t.kind == kind).unwrap_or(false)
    }

    fn match_token(&mut self, kind: TokenKind) -> bool {
        if self.check(kind) {
            self.advance();
            true
        } else {
            false
        }
    }

    fn expect(&mut self, kind: TokenKind) -> NexusResult<Token> {
        match self.advance() {
            Some(token) if token.kind == kind => Ok(token),
            Some(token) => Err(NexusError::UnexpectedToken {
                expected: kind.to_string(),
                found: token.kind.to_string(),
                span: token.span.into(),
            }),
            None => Err(self.unexpected_eof()),
        }
    }

    fn expect_identifier(&mut self) -> NexusResult<Token> {
        match self.advance() {
            Some(token) if token.kind == TokenKind::Identifier => Ok(token),
            Some(token) => Err(NexusError::UnexpectedToken {
                expected: "identifier".into(),
                found: token.kind.to_string(),
                span: token.span.into(),
            }),
            None => Err(self.unexpected_eof()),
        }
    }

    fn expect_semicolon(&mut self) -> NexusResult<()> {
        if self.check(TokenKind::Semicolon) {
            self.advance();
            Ok(())
        } else {
            Err(NexusError::MissingSemicolon {
                span: self.last_span().into(),
            })
        }
    }

    fn peek_operator(&self) -> Option<BinaryOperator> {
        self.peek()
            .and_then(|t| token_to_binary_op(&t.kind))
    }

    fn last_span(&self) -> Span {
        // Return the end of the last token or zero
        self.tokens
            .clone()
            .peek()
            .map(|t| t.span)
            .unwrap_or(Span::new(0, 0))
    }

    fn unexpected_token(&self, expected: &str) -> NexusError {
        match self.peek() {
            Some(token) => NexusError::UnexpectedToken {
                expected: expected.into(),
                found: token.kind.to_string(),
                span: token.span.into(),
            },
            None => self.unexpected_eof(),
        }
    }

    fn unexpected_eof(&self) -> NexusError {
        NexusError::InternalError {
            msg: "Unexpected end of file".into(),
            span: None,
        }
    }
}

// ── Operator precedence ──

fn token_to_binary_op(kind: &TokenKind) -> Option<BinaryOperator> {
    match kind {
        TokenKind::Plus => Some(BinaryOperator::Add),
        TokenKind::Minus => Some(BinaryOperator::Subtract),
        TokenKind::Star => Some(BinaryOperator::Multiply),
        TokenKind::Slash => Some(BinaryOperator::Divide),
        TokenKind::Equals => Some(BinaryOperator::Assign),
        _ => None,
    }
}

fn op_precedence(op: &BinaryOperator) -> u8 {
    match op {
        BinaryOperator::Assign => 1,
        BinaryOperator::Add | BinaryOperator::Subtract => 2,
        BinaryOperator::Multiply | BinaryOperator::Divide => 3,
    }
}
```

crates/nexus-cli/src/main.rs

```rust
use bumpalo::Bump;
use clap::{Parser, Subcommand};
use miette::{IntoDiagnostic, Report};
use nexus_parser::Parser as NexusParser;
use std::path::PathBuf;

#[derive(Parser)]
#[command(name = "nexus", version = "0.1.0", about = "NEXUS Language Compiler & Runtime")]
struct Cli {
    #[command(subcommand)]
    command: Command,
}

#[derive(Subcommand)]
enum Command {
    /// Lex a NEXUS source file and print tokens
    Lex {
        file: PathBuf,
    },
    /// Parse a NEXUS source file and print the AST
    Parse {
        file: PathBuf,
    },
    /// Compile and run a NEXUS source file
    Run {
        file: PathBuf,
    },
}

fn main() -> miette::Result<()> {
    let cli = Cli::parse();

    match cli.command {
        Command::Lex { file } => {
            let source = std::fs::read_to_string(&file)
                .into_diagnostic()
                .map_err(|e: Report| e.with_source_code(file.to_string_lossy().to_string()))?;

            let mut lexer = nexus_lexer::Lexer::new(&source);
            let tokens = lexer.tokenize()
                .map_err(|e| Report::new(e).with_source_code(source))?;

            println!("Tokens ({})", tokens.len());
            for token in &tokens {
                println!("  {:?} '{}' at {:?}", token.kind, token.lexeme, token.span);
            }

            Ok(())
        }

        Command::Parse { file } => {
            let source = std::fs::read_to_string(&file)
                .into_diagnostic()
                .map_err(|e: Report| e.with_source_code(file.to_string_lossy().to_string()))?;

            let arena = Bump::new();
            let mut parser = NexusParser::new(&source, &arena)
                .map_err(|e| Report::new(e).with_source_code(source.clone()))?;

            let program = parser.parse_program()
                .map_err(|e| Report::new(e).with_source_code(source.clone()))?;

            println!("Successfully parsed {} declarations", program.declarations.len());
            for decl in &program.declarations {
                println!("  {:?} (node: {})", std::mem::discriminant(decl), decl.node_id());
            }

            Ok(())
        }

        Command::Run { file } => {
            let source = std::fs::read_to_string(&file)
                .into_diagnostic()
                .map_err(|e: Report| e.with_source_code(file.to_string_lossy().to_string()))?;

            println!("Running: {}", file.display());
            println!("[Runtime execution not yet implemented in v0.1]");
            println!("Source parsed successfully: {} bytes", source.len());

            let arena = Bump::new();
            let mut parser = NexusParser::new(&source, &arena)
                .map_err(|e| Report::new(e).with_source_code(source.clone()))?;
            let _program = parser.parse_program()
                .map_err(|e| Report::new(e).with_source_code(source))?;

            println!("✓ Program structure validated.");

            Ok(())
        }
    }
}
```
