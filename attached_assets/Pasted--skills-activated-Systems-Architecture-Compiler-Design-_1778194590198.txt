use bumpalo::Bump;
use nexus_ast::*;
use nexus_errors::{NexusError, NexusResult, Span};
use nexus_ir::{GraphNode, IrGraph, NodeKind, ValueType};
use std::collections::HashMap;
use tracing::{debug, info, warn};

// ──────────────────────────────────────────────
// Semantic Analyzer
// ──────────────────────────────────────────────

/// The type checker walks the AST and produces:
/// 1. Typed IR graph nodes
/// 2. A symbol table mapping names to types and dimensions
/// 3. Guarantees of unit consistency (no V = A * ohm errors)
pub struct TypeChecker<'arena> {
    arena: &'arena Bump,
    symbol_table: HashMap<String, SymbolInfo>,
    ir_graph: IrGraph,
    current_scope: Scope,
    diagnostics: Vec<NexusError>,
}

#[derive(Debug, Clone)]
struct SymbolInfo {
    value_type: ValueType,
    dimension: Dimension,
    node_id: Option<nexus_ir::NodeId>,
    mutable: bool,
}

#[derive(Debug, Clone, Default)]
struct Scope {
    parent: Option<Box<Scope>>,
    symbols: HashMap<String, SymbolInfo>,
}

impl Scope {
    fn new() -> Self {
        Self::default()
    }

    fn with_parent(parent: Scope) -> Self {
        Self {
            parent: Some(Box::new(parent)),
            symbols: HashMap::new(),
        }
    }

    fn lookup(&self, name: &str) -> Option<&SymbolInfo> {
        self.symbols
            .get(name)
            .or_else(|| self.parent.as_ref().and_then(|p| p.lookup(name)))
    }

    fn insert(&mut self, name: String, info: SymbolInfo) {
        self.symbols.insert(name, info);
    }
}

impl<'arena> TypeChecker<'arena> {
    pub fn new(arena: &'arena Bump) -> Self {
        Self {
            arena,
            symbol_table: HashMap::new(),
            ir_graph: IrGraph::new(),
            current_scope: Scope::new(),
            diagnostics: Vec::new(),
        }
    }

    /// Main entry point: type-check a program and produce an IR graph.
    pub fn check(&mut self, program: &Program) -> Result<IrGraph, Vec<NexusError>> {
        info!("Starting semantic analysis on {} declarations", program.declarations.len());

        for decl in &program.declarations {
            if let Err(e) = self.check_declaration(decl) {
                self.diagnostics.push(e);
            }
        }

        if self.diagnostics.is_empty() {
            debug!("Type checking complete. IR graph has {} nodes.", self.ir_graph.node_count());
            Ok(self.ir_graph.clone())
        } else {
            warn!("Type checking failed with {} errors", self.diagnostics.len());
            Err(self.diagnostics.clone())
        }
    }

    fn check_declaration(&mut self, decl: &AstNode) -> Result<(), NexusError> {
        match decl {
            AstNode::LetBinding(lb) => self.check_let_binding(lb),
            AstNode::EquationDef(ed) => self.check_equation_def(ed),
            AstNode::EntityDef(ed) => self.check_entity_def(ed),
            AstNode::SensorDef(sd) => self.check_sensor_def(sd),
            _ => {
                // Expression statements
                let (_, dim) = self.infer_type(decl)?;
                Ok(())
            }
        }
    }

    fn check_let_binding(&mut self, lb: &LetBinding) -> Result<(), NexusError> {
        debug!("Checking let binding: {}", lb.name);

        // Infer the type and dimension of the value
        let (value_type, value_dim) = self.infer_type(&lb.value)?;

        // If there's a type annotation, verify it matches
        if let Some(annotation) = &lb.type_annotation {
            let (annot_type, annot_dim) = self.infer_type(annotation)?;
            if annot_dim != value_dim {
                return Err(NexusError::UnitMismatch {
                    expected: annot_dim.to_string(),
                    found: value_dim.to_string(),
                    span: lb.span.into(),
                });
            }
        }

        // Create an IR node for the binding
        let ir_node = GraphNode::new(
            NodeKind::Binding(lb.name.clone()),
            value_type,
            value_dim.clone(),
        );

        let node_id = self.ir_graph.add_node(ir_node);

        // Update symbol table
        self.current_scope.insert(
            lb.name.clone(),
            SymbolInfo {
                value_type,
                dimension: value_dim,
                node_id: Some(node_id),
                mutable: false,
            },
        );

        // Add dependencies from the value expression
        self.add_expression_dependencies(node_id, &lb.value, &lb.name)?;

        Ok(())
    }

    fn check_equation_def(&mut self, ed: &EquationDef) -> Result<(), NexusError> {
        debug!("Checking equation definition: {}", ed.name);

        // Equations are symbolic — they define relationships, not values
        // We register them as equation-type nodes
        let eq_node = GraphNode::new(
            NodeKind::Equation(ed.name.clone()),
            ValueType::Equation,
            Dimension::Scalar,
        );

        let node_id = self.ir_graph.add_node(eq_node);

        // The equation body is stored as a symbolic expression
        self.current_scope.insert(
            ed.name.clone(),
            SymbolInfo {
                value_type: ValueType::Equation,
                dimension: Dimension::Scalar,
                node_id: Some(node_id),
                mutable: false,
            },
        );

        self.add_expression_dependencies(node_id, &ed.body, &ed.name)?;

        Ok(())
    }

    fn check_entity_def(&mut self, ed: &EntityDef) -> Result<(), NexusError> {
        debug!("Checking entity definition: {}", ed.name);

        let mem_kind = ed.memory.as_ref().map(|m| match m {
            MemoryKind::Persistent => "persistent",
            MemoryKind::Transient => "transient",
        }).unwrap_or("default");

        let entity_node = GraphNode::new(
            NodeKind::Entity(ed.name.clone()),
            ValueType::Entity,
            Dimension::Scalar,
        );

        let node_id = self.ir_graph.add_node(entity_node);

        self.current_scope.insert(
            ed.name.clone(),
            SymbolInfo {
                value_type: ValueType::Entity,
                dimension: Dimension::Scalar,
                node_id: Some(node_id),
                mutable: true, // Entities can react and change
            },
        );

        // Check skills and reactions in entity scope
        let mut entity_scope = Scope::with_parent(self.current_scope.clone());

        for skill in &ed.skills {
            self.check_skill_in_scope(skill, &mut entity_scope)?;
        }

        for reaction in &ed.reactions {
            self.check_reaction(reaction)?;
        }

        Ok(())
    }

    fn check_skill_in_scope(&mut self, skill: &SkillDef, scope: &mut Scope) -> Result<(), NexusError> {
        let old_scope = std::mem::replace(&mut self.current_scope, scope.clone());
        let result = self.check_skill(skill);
        *scope = std::mem::replace(&mut self.current_scope, old_scope);
        result
    }

    fn check_skill(&mut self, skill: &SkillDef) -> Result<(), NexusError> {
        debug!("Checking skill: {}", skill.name);

        for stmt in &skill.body.statements {
            self.check_declaration(stmt)?;
        }

        Ok(())
    }

    fn check_reaction(&mut self, reaction: &ReactBlock) -> Result<(), NexusError> {
        debug!("Checking reaction block");

        let (trigger_type, trigger_dim) = self.infer_type(&reaction.trigger)?;
        let (body_type, body_dim) = self.infer_type(&reaction.body)?;

        Ok(())
    }

    fn check_sensor_def(&mut self, sd: &SensorDef) -> Result<(), NexusError> {
        debug!("Checking sensor definition: {}", sd.name);

        let sensor_node = GraphNode::new(
            NodeKind::Sensor(sd.name.clone()),
            ValueType::Signal(freq_to_hz(&sd.frequency)),
            Dimension::Scalar,
        );

        let node_id = self.ir_graph.add_node(sensor_node);

        self.current_scope.insert(
            sd.name.clone(),
            SymbolInfo {
                value_type: ValueType::Signal(freq_to_hz(&sd.frequency)),
                dimension: Dimension::Scalar,
                node_id: Some(node_id),
                mutable: false,
            },
        );

        Ok(())
    }

    // ── Type Inference ──

    fn infer_type(&self, expr: &AstNode) -> Result<(ValueType, Dimension), NexusError> {
        match expr {
            AstNode::UnitLiteral(ul) => {
                let dim = ul.unit.dimension();
                Ok((ValueType::Unit(ul.unit.clone()), dim))
            }
            AstNode::IntLiteral(_) => {
                Ok((ValueType::Int, Dimension::Scalar))
            }
            AstNode::FloatLiteral(_) => {
                Ok((ValueType::Float, Dimension::Scalar))
            }
            AstNode::StringLiteral(_) => {
                Ok((ValueType::String, Dimension::Scalar))
            }
            AstNode::Identifier(id) => {
                match self.current_scope.lookup(&id.name) {
                    Some(sym) => Ok((sym.value_type.clone(), sym.dimension.clone())),
                    None => Err(NexusError::UndefinedIdentifier {
                        name: id.name.clone(),
                        span: id.span.into(),
                    }),
                }
            }
            AstNode::BinaryOp(bin) => {
                let (left_type, left_dim) = self.infer_type(&bin.left)?;
                let (right_type, right_dim) = self.infer_type(&bin.right)?;

                match bin.op {
                    BinaryOperator::Multiply => {
                        // Engineering dimension multiplication
                        let result_dim = multiply_dimensions(&left_dim, &right_dim)
                            .unwrap_or_else(|| {
                                self.diagnostics.push(NexusError::DimensionalError {
                                    msg: format!(
                                        "Cannot multiply {} by {}",
                                        left_dim, right_dim
                                    ),
                                    span: bin.span.into(),
                                });
                                Dimension::Scalar
                            });
                        Ok((combine_types(&left_type, &right_type), result_dim))
                    }
                    BinaryOperator::Divide => {
                        let result_dim = divide_dimensions(&left_dim, &right_dim)
                            .unwrap_or(Dimension::Scalar);
                        Ok((combine_types(&left_type, &right_type), result_dim))
                    }
                    BinaryOperator::Add | BinaryOperator::Subtract => {
                        // Must be same dimension
                        if left_dim != right_dim {
                            return Err(NexusError::UnitMismatch {
                                expected: left_dim.to_string(),
                                found: right_dim.to_string(),
                                span: bin.span.into(),
                            });
                        }
                        Ok((combine_types(&left_type, &right_type), left_dim))
                    }
                    BinaryOperator::Assign => {
                        Ok((left_type, left_dim))
                    }
                }
            }
            AstNode::Block(block) => {
                if let Some(last) = block.statements.last() {
                    self.infer_type(last)
                } else {
                    Ok((ValueType::Unit, Dimension::Scalar))
                }
            }
            _ => Ok((ValueType::Unknown, Dimension::Scalar)),
        }
    }

    fn add_expression_dependencies(
        &mut self,
        target_id: nexus_ir::NodeId,
        expr: &AstNode,
        context: &str,
    ) -> Result<(), NexusError> {
        match expr {
            AstNode::Identifier(id) => {
                if let Some(sym) = self.current_scope.lookup(&id.name) {
                    if let Some(source_id) = sym.node_id {
                        self.ir_graph.add_dependency(target_id, source_id);
                    }
                }
            }
            AstNode::BinaryOp(bin) => {
                self.add_expression_dependencies(target_id, &bin.left, context)?;
                self.add_expression_dependencies(target_id, &bin.right, context)?;
            }
            _ => {}
        }
        Ok(())
    }
}

// ── Dimensional Analysis Helpers ──

fn multiply_dimensions(a: &Dimension, b: &Dimension) -> Option<Dimension> {
    use Dimension::*;
    match (a, b) {
        (Voltage, Current) => Some(Power),    // V * A = W
        (Current, Voltage) => Some(Power),
        (Current, Resistance) => Some(Voltage), // A * Ω = V
        (Resistance, Current) => Some(Voltage),
        (Scalar, x) | (x, Scalar) => Some(x.clone()),
        (x, y) if x == y => Some(x.clone()),
        _ => None,
    }
}

fn divide_dimensions(a: &Dimension, b: &Dimension) -> Option<Dimension> {
    use Dimension::*;
    match (a, b) {
        (Voltage, Current) => Some(Resistance), // V / A = Ω
        (Power, Current) => Some(Voltage),       // W / A = V
        (Power, Voltage) => Some(Current),       // W / V = A
        (x, Scalar) => Some(x.clone()),
        (x, y) if x == y => Some(Scalar),
        _ => None,
    }
}

fn combine_types(a: &ValueType, b: &ValueType) -> ValueType {
    use ValueType::*;
    match (a, b) {
        (Int, Int) => Int,
        (Float, Float) => Float,
        (Unit(_), Unit(_)) => a.clone(),
        (Int, Float) | (Float, Int) => Float,
        _ => a.clone(),
    }
}

fn freq_to_hz(freq: &Option<UnitLiteral>) -> f64 {
    freq.as_ref().map(|f| f.value).unwrap_or(1.0)
}

// ──────────────────────────────────────────────
// Graph Node
// ──────────────────────────────────────────────

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct GraphNode {
    pub id: NodeId,
    pub kind: NodeKind,
    pub value_type: ValueType,
    pub dimension: Dimension,
    pub value: Option<RuntimeValue>,
    pub dirty: bool,

            kind,
            value_type,

#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
pub enum RuntimeValue {
    Int(i64),
    Float(f64),
    String(String),
    Bool(bool),
    Unit { value: f64, unit: EngineeringUnit },
}

impl RuntimeValue {
    pub fn as_float(&self) -> Option<f64> {
        match self {
            RuntimeValue::Int(i) => Some(*i as f64),
            RuntimeValue::Float(f) => Some(*f),
            RuntimeValue::Unit { value, .. } => Some(*value),
            _ => None,

// ──────────────────────────────────────────────
// IR Graph — The Reactive Dependency Graph
// ──────────────────────────────────────────────

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct IrGraph {
    nodes: HashMap<NodeId, GraphNode>,
    /// edges[target] = set of source nodes it depends on
    dependencies: HashMap<NodeId, HashSet<NodeId>>,
    /// reverse edges: dependents[source] = nodes that depend on source
    dependents: HashMap<NodeId, HashSet<NodeId>>,
    /// Topological order for computation
    eval_order: Vec<NodeId>,
}

impl IrGraph {
    pub fn new() -> Self {
        Self {
            nodes: HashMap::new(),
            dependencies: HashMap::new(),
            dependents: HashMap::new(),
            eval_order: Vec::new(),
        }
    }

    pub fn add_node(&mut self, node: GraphNode) -> NodeId {
        let id = node.id;
        self.nodes.insert(id, node);
        self.dependencies.entry(id).or_insert_with(HashSet::new);
        self.dependents.entry(id).or_insert_with(HashSet::new);
        id
    }

    pub fn add_dependency(&mut self, target: NodeId, source: NodeId) {
        self.dependencies
            .entry(target)
            .or_insert_with(HashSet::new)
            .insert(source);
        self.dependents
            .entry(source)
            .or_insert_with(HashSet::new)
            .insert(target);
    }

    pub fn mark_dirty(&mut self, node_id: NodeId) {
        // Mark node and all transitive dependents as dirty
        let mut queue = vec![node_id];

        while let Some(current) = queue.pop() {
            if let Some(node) = self.nodes.get_mut(&current) {
                node.dirty = true;
            }

            if let Some(deps) = self.dependents.get(&current) {
                for &dep in deps {
                    if let Some(node) = self.nodes.get(&dep) {
                        if !node.dirty {
                            queue.push(dep);
                        }
                    }
                }
            }
        }

        // Recompute topological order
        self.recompute_order();
    }

    pub fn set_value(&mut self, node_id: NodeId, value: RuntimeValue) {
        if let Some(node) = self.nodes.get_mut(&node_id) {
            node.value = Some(value);
            node.dirty = true;
        }
        self.mark_dirty(node_id);
    }

    fn recompute_order(&mut self) {
        // Topological sort of dirty nodes
        let mut visited = HashSet::new();
        let mut order = Vec::new();

        // Find all dirty nodes
        let dirty_nodes: Vec<NodeId> = self
            .nodes
            .iter()
            .filter(|(_, node)| node.dirty)
            .map(|(id, _)| *id)
            .collect();

        for node_id in &dirty_nodes {
            self.topo_visit(*node_id, &mut visited, &mut order);
        }

        order.reverse();
        self.eval_order = order;
    }

    fn topo_visit(&self, node_id: NodeId, visited: &mut HashSet<NodeId>, order: &mut Vec<NodeId>) {
        if visited.contains(&node_id) {
            return;
        }
        visited.insert(node_id);

        if let Some(deps) = self.dependencies.get(&node_id) {
            for &dep in deps {
                self.topo_visit(dep, visited, order);
            }
        }

        order.push(node_id);
    }

    pub fn nodes(&self) -> &HashMap<NodeId, GraphNode> {
        &self.nodes
    }

    pub fn node_count(&self) -> usize {
        self.nodes.len()
    }

    pub fn get_node(&self, id: &NodeId) -> Option<&GraphNode> {
        self.nodes.get(id)
    }

    pub fn get_node_mut(&mut self, id: &NodeId) -> Option<&mut GraphNode> {
        self.nodes.get_mut(id)
    }

    pub fn eval_order(&self) -> &[NodeId] {
        &self.eval_order
    }

    pub fn dependencies_of(&self, node_id: &NodeId) -> Vec<NodeId> {
        self.dependencies
            .get(node_id)
            .map(|set| set.iter().copied().collect())
            .unwrap_or_default()
    }
}

use nexus_ast::Dimension;
use nexus_ir::*;
use std::collections::HashMap;
use std::sync::Arc;
use tokio::sync::RwLock;
use tracing::{debug, info, trace};

// ──────────────────────────────────────────────
// Reactive Runtime Engine
// ──────────────────────────────────────────────

/// The core reactive execution engine.
/// Manages the dependency graph, change propagation, and value computation.
pub struct ReactiveRuntime {
    graph: Arc<RwLock<IrGraph>>,
    /// Callbacks registered for specific node changes
    watchers: HashMap<NodeId, Vec<Box<dyn Fn(&RuntimeValue) + Send + Sync>>>,
}

impl ReactiveRuntime {
    pub fn new(graph: IrGraph) -> Self {
        Self {
            graph: Arc::new(RwLock::new(graph)),
            watchers: HashMap::new(),
        }
    }

    /// Execute all dirty nodes in topological order.
    /// This is the core reactive propagation loop.
    pub async fn propagate(&mut self) -> Result<usize, String> {
        let graph = self.graph.read().await;
        let eval_order = graph.eval_order().to_vec();
        drop(graph);

        let mut recomputed = 0;

        for node_id in &eval_order {
            let mut graph = self.graph.write().await;
            if let Some(node) = graph.get_node(node_id) {
                if node.dirty {
                    trace!("Recomputing node: {:?}", node.kind);
                    if let Ok(new_value) = self.evaluate_node(*node_id).await {
                        // Notify watchers
                        if let Some(watchers) = self.watchers.get(node_id) {
                            for watcher in watchers {
                                watcher(&new_value);
                            }
                        }
                        if let Some(node) = graph.get_node_mut(node_id) {
                            node.value = Some(new_value);
                            node.dirty = false;
                            recomputed += 1;
                        }
                    }
                }
            }
        }

        info!("Propagation complete: {} nodes recomputed", recomputed);
        Ok(recomputed)
    }

    /// Evaluate a single node based on its kind and dependencies' values.
    async fn evaluate_node(&self, node_id: NodeId) -> Result<RuntimeValue, String> {
        let graph = self.graph.read().await;
        let node = graph
            .get_node(&node_id)
            .cloned()
            .ok_or_else(|| format!("Node {:?} not found", node_id))?;

        match &node.kind {
            NodeKind::Literal(lit) => Ok(literal_to_value(lit)),

            NodeKind::Operation(op) => {
                let deps = graph.dependencies_of(&node_id);
                let dep_values: Vec<RuntimeValue> = deps
                    .iter()
                    .filter_map(|id| graph.get_node(id))
                    .filter_map(|n| n.value.clone())
                    .collect();

                Self::execute_operation(op, &dep_values)
            }

            NodeKind::Binding(_) => {
                // A binding's value comes from its dependency (the expression it was assigned)
                let deps = graph.dependencies_of(&node_id);
                if let Some(dep_id) = deps.first() {
                    graph
                        .get_node(dep_id)
                        .and_then(|n| n.value.clone())
                        .ok_or_else(|| "Binding dependency has no value".to_string())
                } else if let Some(value) = &node.value {
                    Ok(value.clone())
                } else {
                    Err("Binding has no computed value".to_string())
                }
            }

            NodeKind::Equation(_) | NodeKind::Entity(_) | NodeKind::Sensor(_) => {
                // These are structural nodes — they hold metadata
                Ok(node
                    .value
                    .clone()
                    .unwrap_or(RuntimeValue::String(node.kind_summary())))
            }

            NodeKind::Reaction(_) => {
                Ok(RuntimeValue::Bool(true)) // Placeholder for reaction evaluation
            }

            NodeKind::Prediction(_) => {
                Ok(RuntimeValue::String("prediction_pending".to_string()))
            }
        }
    }

    /// Execute arithmetic operations on runtime values
    fn execute_operation(op: &OperationKind, values: &[RuntimeValue]) -> Result<RuntimeValue, String> {
        if values.len() < 2 {
            return Err(format!("Operation {:?} requires at least 2 operands", op));
        }

        let a = &values[0];
        let b = &values[1];

        let a_val = a.as_float().ok_or("Operand is not numeric")?;
        let b_val = b.as_float().ok_or("Operand is not numeric")?;

        let result_val = match op {
            OperationKind::Add => a_val + b_val,
            OperationKind::Subtract => a_val - b_val,
            OperationKind::Multiply => a_val * b_val,
            OperationKind::Divide => {
                if b_val == 0.0 {
                    return Err("Division by zero".to_string());
                }
                a_val / b_val
            }
            OperationKind::Negate => -a_val,
        };

        // Preserve unit information if present
        match (a, b) {
            (RuntimeValue::Unit { unit: ua, .. }, RuntimeValue::Unit { unit: ub, .. }) => {
                let combined_dim = match op {
                    OperationKind::Multiply => {
                        // V * A = W, etc.
                        multiply_units(ua, ub)
                    }
                    OperationKind::Divide => divide_units(ua, ub),
                    _ => Some(ua.clone()),
                };

                match combined_dim {
                    Some(unit) => Ok(RuntimeValue::Unit {
                        value: result_val,
                        unit,
                    }),
                    None => Ok(RuntimeValue::Float(result_val)),
                }
            }
            (RuntimeValue::Unit { unit, .. }, _) => Ok(RuntimeValue::Unit {
                value: result_val,
                unit: unit.clone(),
            }),
            (_, RuntimeValue::Unit { unit, .. }) => Ok(RuntimeValue::Unit {
                value: result_val,
                unit: unit.clone(),
            }),
            _ => {
                if result_val.fract() == 0.0 && result_val <= i64::MAX as f64 {
                    Ok(RuntimeValue::Int(result_val as i64))
                } else {
                    Ok(RuntimeValue::Float(result_val))
                }
            }
        }
    }

    /// Watch a node for changes — reactive callback pattern
    pub fn watch<F>(&mut self, node_id: NodeId, callback: F)
    where
        F: Fn(&RuntimeValue) + Send + Sync + 'static,
    {
        self.watchers
            .entry(node_id)
            .or_insert_with(Vec::new)
            .push(Box::new(callback));
    }

    /// Set an external value (like sensor input) and trigger propagation
    pub async fn set_external(&mut self, node_id: NodeId, value: RuntimeValue) {
        let mut graph = self.graph.write().await;
        graph.set_value(node_id, value);
        drop(graph);
        let _ = self.propagate().await;
    }

    /// Get the current value of any node
    pub async fn get_value(&self, node_id: &NodeId) -> Option<RuntimeValue> {
        let graph = self.graph.read().await;
        graph.get_node(node_id).and_then(|n| n.value.clone())
    }

    /// Print the current state of the entire graph (debugging)
    pub async fn dump_state(&self) -> String {
        let graph = self.graph.read().await;
        let mut output = String::from("\n═══ GRAPH STATE ═══\n");
        for (id, node) in graph.nodes() {
            let val_str = node
                .value
                .as_ref()
                .map(|v| format!("{:?}", v))
                .unwrap_or_else(|| "<empty>".to_string());
            output.push_str(&format!(
                "  {:?} [{}] = {}  (dirty: {})\n",
                node.kind,
                node.dimension,
                val_str,
                node.dirty
            ));
        }
        output
    }
}

pub struct NexusVm {

        Ok(())

        let rt = self.runtime.lock().await;
        rt.dump_state().await
    }

    /// Shutdown the VM gracefully
    pub async fn shutdown(&mut self) {
        info!("Shutting down NEXUS VM...");
        self.running = false;
    }

// ──────────────────────────────────────────────
// NEXUS Mathematical Engine (§11)

/// Symbolic equation solver. Given V = I * R and two known values,
/// solves for the third.
pub struct EquationSolver;

impl EquationSolver {
    /// Solve an equation given known variables.
    /// 
    /// Example:
    /// equation OhmLaw { V = I * R }
    /// known: { "V": 12V, "I": 3A }
    /// solves for R = 4ohm
    pub fn solve_linear(
        equation: &str, // symbolic form: "V = I * R"
        known: &[(&str, RuntimeValue)],
        unknown: &str,
    ) -> Option<f64> {
        debug!("Solving equation '{}' for '{}'", equation, unknown);

        // Parse the equation form
        let parts: Vec<&str> = equation.split('=').collect();
        if parts.len() != 2 {
            return None;
        }

        let left = parts[0].trim();
        let right = parts[1].trim();

        // Simple linear solver for V = I * R form
        Self::solve_linear_multiplicative(left, right, known, unknown)
    }

    fn solve_linear_multiplicative(
        left: &str,
        right: &str,
        known: &[(&str, RuntimeValue)],
        unknown: &str,
    ) -> Option<f64> {
        // Handle: unknown = expr  or  expr = unknown
        let known_map: std::collections::HashMap<&str, f64> = known
            .iter()
            .filter_map(|(name, val)| val.as_float().map(|v| (*name, v)))
            .collect();

        // Very simple case: if right is "I * R", and we know I, solve for R
        if right.contains('*') {
            let factors: Vec<&str> = right.split('*').map(|s| s.trim()).collect();
            if factors.len() == 2 {
                let a = factors[0];
                let b = factors[1];

                if a == unknown {
                    // unknown = left / b
                    let left_val = known_map.get(left).or_else(|| known_map.get("V"))?;
                    let b_val = known_map.get(b)?;
                    return Some(left_val / b_val);
                } else if b == unknown {
                    // unknown = left / a
                    let left_val = known_map.get(left).or_else(|| known_map.get("V"))?;
                    let a_val = known_map.get(a)?;
                    return Some(left_val / a_val);
                } else if left == unknown {
                    // unknown = a * b
                    let a_val = known_map.get(a)?;
                    let b_val = known_map.get(b)?;
                    return Some(a_val * b_val);
                }
            }
        }

        None
    }

    /// Rearrange V = I * R to solve for any variable symbolically
    pub fn rearrange(equation: &str, solve_for: &str) -> Option<String> {
        let parts: Vec<&str> = equation.split('=').collect();
        if parts.len() != 2 {
            return None;
        }

        let left = parts[0].trim();
        let right = parts[1].trim();

        if left == solve_for {
            Some(format!("{} = {}", solve_for, right))
        } else if right.contains(solve_for) && right.contains('*') {
            let factors: Vec<&str> = right.split('*').map(|s| s.trim()).collect();
            if factors.len() == 2 {
                let other = if factors[0] == solve_for {
                    factors[1]
                } else {
                    factors[0]
                };
                Some(format!("{} = {} / {}", solve_for, left, other))
            } else {
                None
            }
        } else {
            None
        }
    }
}

/// Matrix and tensor operations for AI/ML workloads
pub struct TensorEngine;

impl TensorEngine {
    pub fn matrix_multiply(a: &[Vec<f64>], b: &[Vec<f64>]) -> Option<Vec<Vec<f64>>> {
        if a.is_empty() || b.is_empty() || a[0].len() != b.len() {
            return None;
        }

        let rows = a.len();
        let cols = b[0].len();
        let inner = b.len();

        let na_rows = rows;
        let na_cols = inner;

        let nb_rows = inner;
        let nb_cols = cols;

        let mut result = vec![vec![0.0; cols]; rows];

        for i in 0..rows {
            for j in 0..cols {
                let mut sum = 0.0;
                for k in 0..inner {
                    sum += a[i][k] * b[k][j];
                }
                result[i][j] = sum;
            }
        }

        Some(result)
    }

    pub fn dot_product(a: &[f64], b: &[f64]) -> Option<f64> {
        if a.len() != b.len() {
            return None;
        }
        Some(a.iter().zip(b.iter()).map(|(x, y)| x * y).sum())
    }
}

fn read_source(file: &PathBuf) -> miette::Result<String> {
    let source = std::fs::read_to_string(file)
        .into_diagnostic()
        .map_err(|e: Report| {
            e.with_source_code(NamedSource::new(
                file.to_string_lossy(),
                "could not read file",
            ))
        })?;
    Ok(source)
}

fn cmd_lex(file: PathBuf) -> miette::Result<()> {
    let source = read_source(&file)?;
    let mut lexer = Lexer::new(&source);

    match lexer.tokenize() {
        Ok(tokens) => {
            println!("╔══════════════════════════════════════════╗");
            println!("║  NEXUS Lexer — Token Stream              ║");
            println!("╚══════════════════════════════════════════╝");
            println!("  {} tokens:\n", tokens.len());

            for token in &tokens {
                println!(
                    "  [{:>3}:{:<3}] {:?}  '{}'",
                    token.span.start,
                    token.span.end,
                    token.kind,
                    token.lexeme
                );
            }
            Ok(())
        }
        Err(e) => {
            Err(Report::new(e).with_source_code(source))
        }

fn print_ast_node(node: &nexus_ast::AstNode, depth: usize) {
    let indent = "  ".repeat(depth);
    match node {

            println!("{}LetBinding: {}", indent, lb.name);
            println!("{}  value:", indent);
            print_ast_node(&lb.value, depth + 2);
        }

            println!("{}EquationDef: {}", indent, ed.name);
        }

            println!("{}SensorDef: {} (pin: {})", indent, sd.name, sd.pin);
        }
        nexus_ast::AstNode::UnitLiteral(ul) => {
            println!("{}UnitLiteral: {}{:?}", indent, ul.value, ul.unit);
        }
        nexus_ast::AstNode::Identifier(id) => {
            println!("{}Identifier: {}", indent, id.name);
        }
        nexus_ast::AstNode::BinaryOp(bin) => {
            println!("{}BinaryOp: {:?}", indent, bin.op);
            print_ast_node(&bin.left, depth + 1);
            print_ast_node(&bin.right, depth + 1);
        }
        _ => println!("{}{:?}", indent, std::mem::discriminant(node)),
    }
}

fn cmd_check(file: PathBuf) -> miette::Result<()> {
    let source = read_source(&file)?;
    let arena = Bump::new();

    let mut parser = NexusParser::new(&source, &arena)
        .map_err(|e| Report::new(e).with_source_code(source.clone()))?;

    let program = parser.parse_program()
        .map_err(|e| Report::new(e).with_source_code(source.clone()))?;

    let mut typeck = TypeChecker::new(&arena);

    match typeck.check(&program) {
        Ok(ir_graph) => {
            println!("╔══════════════════════════════════════════╗");
            println!("║  NEXUS Type Checker — SUCCESS            ║");
            println!("╚══════════════════════════════════════════╝");
            println!("  IR Graph: {} nodes", ir_graph.node_count());
            println!("  No errors found. All units consistent.");
        }
        Err(errors) => {
            eprintln!("╔══════════════════════════════════════════╗");
            eprintln!("║  NEXUS Type Checker — {} ERROR(S)        ║", errors.len());
            eprintln!("╚══════════════════════════════════════════╝");
            for error in &errors {
                eprintln!("  {}", error);
            }
            std::process::exit(1);
        }
    }

    Ok(())
}

async fn cmd_run(file: PathBuf, verbose: bool) -> miette::Result<()> {
    let source = read_source(&file)?;
    let arena = Bump::new();

    let mut parser = NexusParser::new(&source, &arena)
        .map_err(|e| Report::new(e).with_source_code(source.clone()))?;

    let program = parser.parse_program()
        .map_err(|e| Report::new(e).with_source_code(source.clone()))?;

    let mut typeck = TypeChecker::new(&arena);
    let ir_graph = typeck.check(&program)
        .map_err(|errors| {
            let msg = errors.iter().map(|e| e.to_string()).collect::<Vec<_>>().join("\n");
            Report::msg(msg)
        })?;

    println!("╔══════════════════════════════════════════╗");
    println!("║  NEXUS Runtime — Executing               ║");
    println!("╚══════════════════════════════════════════╝");
    println!("  IR Graph: {} nodes ready", ir_graph.node_count());

    let mut vm = NexusVm::new(ir_graph);
    vm.boot().await.map_err(|e| Report::msg(e))?;
    vm.tick().await.map_err(|e| Report::msg(e))?;

    if verbose {
        println!("{}", vm.inspect().await);
    }

    println!("\n  ✓ Execution complete.");

    vm.shutdown().await;
    Ok(())
}

async fn cmd_repl() -> miette::Result<()> {
    use std::io::{self, Write};

    println!("╔══════════════════════════════════════════╗");
    println!("║  NEXUS REPL v0.1.0                       ║");
    println!("║  Type .exit to quit, .help for commands  ║");
    println!("╚══════════════════════════════════════════╝\n");

    let mut vm: Option<NexusVm> = None;
    let arena = Bump::new();

    loop {
        print!("nexus> ");
        io::stdout().flush().unwrap();

        let mut input = String::new();
        io::stdin().read_line(&mut input).unwrap();
        let input = input.trim();

        match input {
            ".exit" | ".quit" => {
                if let Some(mut vm) = vm.take() {
                    vm.shutdown().await;
                }
                println!("Goodbye.");
                break;
            }
            ".help" => {
                println!("  .exit      — Exit the REPL");
                println!("  .help      — Show this help");
                println!("  .state     — Show current graph state");
                println!("  <code>     — Parse and execute NEXUS code");
            }
            ".state" => {
                if let Some(ref vm) = vm {
                    println!("{}", vm.inspect().await);
                } else {
                    println!("  No program loaded.");
                }
            }
            code if !code.is_empty() => {
                let mut parser = match NexusParser::new(code, &arena) {
                    Ok(p) => p,
                    Err(e) => {
                        eprintln!("  Parse error: {}", e);
                        continue;
                    }
                };

                match parser.parse_program() {
                    Ok(program) => {
                        let mut typeck = TypeChecker::new(&arena);
                        match typeck.check(&program) {
                            Ok(ir_graph) => {
                                let mut new_vm = NexusVm::new(ir_graph);
                                match new_vm.boot().await {
                                    Ok(()) => {
                                        let _ = new_vm.tick().await;
                                        println!("  ✓ Evaluated. {} nodes.", 
                                            new_vm.inspect().await.lines().count() - 2);
                                        vm = Some(new_vm);
                                    }
                                    Err(e) => eprintln!("  Runtime error: {}", e),
                                }
                            }
                            Err(errors) => {
                                for error in &errors {
                                    eprintln!("  Error: {}", error);
                                }
                            }
                        }
                    }
                    Err(e) => {
                        eprintln!("  Parse error: {}", e);
                    }
                }
            }
            _ => {}
        }
    }

    Ok(())
}
